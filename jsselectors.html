JSON Concepts

JSON (JavaScript Object Notation) is a lightweight data interchange format that is easy for humans to read and write, and easy for machines to parse and generate.

It is widely used for data exchange between a server and a web application. JSON is compatible with many programming languages, including JavaScript.

JSON Examples

Example 1: Fetching JSON Data using Fetch API

Fetch JSON data from a local file using the fetch API. 

fetch('data.json')
  .then(response => response.json())
  .then(data => {
    console.log(data); // Handle JSON data here
    renderData(data); // Example function to render data
  })
  .catch(error => {
    console.error('Error fetching data:', error);
  });

function renderData(data) {
  // Example: Display data in HTML
  const container = document.getElementById('json-data');
  container.innerHTML = JSON.stringify(data, null, 2);
}

Example 2: Fetching JSON Data using XMLHttpRequest

Use XMLHttpRequest to fetch JSON data asynchronously. 

var xhr = new XMLHttpRequest();
xhr.open('GET', 'data.json', true);
xhr.onreadystatechange = function() {
  if (xhr.readyState === XMLHttpRequest.DONE) {
    if (xhr.status === 200) {
      var data = JSON.parse(xhr.responseText);
      console.log(data); // Handle JSON data here
      renderData(data); // Example function to render data
    } else {
      console.error('Error fetching data:', xhr.statusText);
    }
  }
};
xhr.send();

function renderData(data) {
  // Example: Display data in HTML
  const container = document.getElementById('json-data');
  container.innerHTML = JSON.stringify(data, null, 2);
}

Example 3: Error Handling with Fetch API

Handle errors when fetching JSON data using the Fetch API. 

fetch('data.json')
  .then(response => {
    if (!response.ok) {
      throw new Error('Network response was not ok');
    }
    return response.json();
  })
  .then(data => {
    console.log(data); // Handle JSON data here
    renderData(data); // Example function to render data
  })
  .catch(error => {
    console.error('Error fetching data:', error.message);
  });

function renderData(data) {
  // Example: Display data in HTML
  const container = document.getElementById('json-data');
  container.innerHTML = JSON.stringify(data, null, 2);
}

Example 4: Fetch API with Custom Options

Customize fetch options, such as headers and methods. 

fetch('https://api.example.com/data', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_TOKEN_HERE'
  },
  body: JSON.stringify({ key: 'value' })
})
.then(response => response.json())
.then(data => {
  console.log(data); // Handle JSON data here
  renderData(data); // Example function to render data
})
.catch(error => {
  console.error('Error fetching data:', error);
});

function renderData(data) {
  // Example: Display data in HTML
  const container = document.getElementById('json-data');
  container.innerHTML = JSON.stringify(data, null, 2);
}
